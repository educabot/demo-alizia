## Descripción

Usecase para editar una planificación existente: título, objetivo, actividades y fuente.

## Implementación

```go
// src/core/usecases/update_lesson_plan.go

type UpdateLessonPlanInput struct {
    Title           *string                 `json:"title"`
    Objective       *string                 `json:"objective"`
    Activities      *[]LessonPlanActivityIn `json:"activities"`
    SourceType      *entities.SourceType    `json:"source_type"`
    SourceReference *string                 `json:"source_reference"`
}

func (uc *UpdateLessonPlanUsecase) Execute(userID uint, planID uint, input UpdateLessonPlanInput) (*entities.TeacherLessonPlan, error) {
    plan, _ := uc.lessonPlanProvider.GetByID(planID)

    // Auth: solo el docente asignado
    cs, _ := uc.courseSubjectProvider.GetByID(plan.CourseSubjectID)
    if cs.TeacherID != userID {
        return nil, errors.NewForbidden("only assigned teacher")
    }

    // Update fields (merge parcial)
    if input.Title != nil {
        plan.Title = *input.Title
    }
    if input.Objective != nil {
        plan.Objective = input.Objective
    }
    if input.SourceType != nil {
        plan.SourceType = input.SourceType
    }
    if input.SourceReference != nil {
        plan.SourceReference = input.SourceReference
    }

    uc.lessonPlanProvider.Update(plan)

    // Si se enviaron activities, replace all
    if input.Activities != nil {
        if err := uc.validateActivities(*input.Activities, plan.OrganizationID); err != nil {
            return nil, err
        }
        // Delete existing
        uc.lessonPlanProvider.DeleteActivitiesByPlan(planID)
        // Insert new
        for _, a := range *input.Activities {
            activity := &entities.LessonPlanActivity{
                TeacherLessonPlanID: planID,
                Moment:             a.Moment,
                ActivityID:         a.ActivityID,
                CustomNote:         a.CustomNote,
                Order:              a.Order,
            }
            uc.lessonPlanProvider.AddActivity(activity)
        }
    }

    return uc.lessonPlanProvider.GetByID(planID)
}
```

## Notas

- El PATCH es parcial: solo actualiza los campos enviados
- Las activities se reemplazan completas (delete + insert) cuando se incluyen en el request
- Se reutiliza la misma validación de momentos del create
