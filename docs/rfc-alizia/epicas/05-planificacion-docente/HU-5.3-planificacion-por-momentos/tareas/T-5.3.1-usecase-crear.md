# T-5.3.1: Usecase — crear planificación

**Historia:** HU-5.3 — Planificación por momentos
**Tipo:** Backend
**Estado:** ⬜ Pendiente

## Descripción

Usecase para crear una planificación docente con validación de momentos, actividades y fuentes.

## Implementación

```go
// src/core/usecases/create_lesson_plan.go

type CreateLessonPlanInput struct {
    CourseSubjectID  uint                    `json:"course_subject_id" validate:"required"`
    CoordDocClassID  uint                    `json:"coord_doc_class_id" validate:"required"`
    Title            string                  `json:"title" validate:"required"`
    Objective        *string                 `json:"objective"`
    Activities       []LessonPlanActivityIn  `json:"activities" validate:"required,min=3"`
    SourceType       *entities.SourceType    `json:"source_type"`
    SourceReference  *string                 `json:"source_reference"`
}

type LessonPlanActivityIn struct {
    Moment     entities.ClassMoment `json:"moment" validate:"required"`
    ActivityID uint                 `json:"activity_id" validate:"required"`
    CustomNote *string              `json:"custom_note"`
    Order      int                  `json:"order"`
}

func (uc *CreateLessonPlanUsecase) Execute(userID uint, input CreateLessonPlanInput) (*entities.TeacherLessonPlan, error) {
    // 1. Verificar que el user es el docente del course_subject
    cs, _ := uc.courseSubjectProvider.GetByID(input.CourseSubjectID)
    if cs.TeacherID != userID {
        return nil, errors.NewForbidden("only assigned teacher can create plans")
    }

    // 2. Verificar que el coord_doc_class pertenece a un documento published
    class, _ := uc.coordDocClassProvider.GetByID(input.CoordDocClassID)
    doc, _ := uc.coordDocProvider.GetByID(class.CoordinationDocumentSubject.CoordinationDocumentID)
    if doc.Status != entities.CoordDocStatusPublished {
        return nil, errors.NewBadRequest("document must be published")
    }

    // 3. Validar momentos y actividades
    if err := uc.validateActivities(input.Activities, cs.OrganizationID); err != nil {
        return nil, err
    }

    // 4. Validar fuente
    if input.SourceType != nil && *input.SourceType == entities.SourceTypeResource {
        if input.SourceReference == nil || *input.SourceReference == "" {
            return nil, errors.NewValidation("source_reference required for resource type")
        }
    }

    // 5. Crear plan
    plan := &entities.TeacherLessonPlan{
        OrganizationID:         cs.OrganizationID,
        CourseSubjectID:        input.CourseSubjectID,
        CoordinationDocumentID: doc.ID,
        CoordDocClassID:        input.CoordDocClassID,
        ClassNumber:            class.ClassNumber,
        Title:                  input.Title,
        Objective:              input.Objective,
        Status:                 entities.LessonPlanStatusPending,
        SourceType:             input.SourceType,
        SourceReference:        input.SourceReference,
    }
    if err := uc.lessonPlanProvider.Create(plan); err != nil {
        return nil, err
    }

    // 6. Crear activities
    for _, a := range input.Activities {
        activity := &entities.LessonPlanActivity{
            TeacherLessonPlanID: plan.ID,
            Moment:             a.Moment,
            ActivityID:         a.ActivityID,
            CustomNote:         a.CustomNote,
            Order:              a.Order,
        }
        uc.lessonPlanProvider.AddActivity(activity)
    }

    return uc.lessonPlanProvider.GetByID(plan.ID)
}

func (uc *CreateLessonPlanUsecase) validateActivities(activities []LessonPlanActivityIn, orgID uint) error {
    opening := filterByMoment(activities, entities.ClassMomentOpening)
    development := filterByMoment(activities, entities.ClassMomentDevelopment)
    closing := filterByMoment(activities, entities.ClassMomentClosing)

    if len(opening) != 1 {
        return errors.NewValidation("exactly 1 opening activity required")
    }
    if len(closing) != 1 {
        return errors.NewValidation("exactly 1 closing activity required")
    }
    if len(development) < 1 {
        return errors.NewValidation("at least 1 development activity required")
    }

    // Check max desarrollo from org config
    org, _ := uc.orgProvider.GetByID(orgID)
    maxDev := org.Config.DesarrolloMaxActivities // default 3
    if len(development) > maxDev {
        return errors.NewValidation(fmt.Sprintf("max %d development activities", maxDev))
    }

    // Verify each activity exists and moment matches
    for _, a := range activities {
        act, err := uc.activityProvider.GetByID(a.ActivityID)
        if err != nil {
            return errors.NewValidation(fmt.Sprintf("activity %d not found", a.ActivityID))
        }
        if act.Moment != a.Moment {
            return errors.NewValidation(fmt.Sprintf("activity %d is %s, not %s", a.ActivityID, act.Moment, a.Moment))
        }
    }
    return nil
}
```

## Notas

- El plan se crea siempre en status `pending`
- La validación de actividades cruza con el catálogo para verificar que cada actividad es del momento correcto
- `desarrollo_max_activities` viene de la config de la organización
