## Descripción

Usecase que arma el contexto y llama a Azure OpenAI para generar la propuesta de clase.

## Implementación

```go
// src/core/usecases/generate_lesson_proposal.go

func (uc *GenerateLessonProposalUsecase) Execute(userID uint, planID uint) (*entities.TeacherLessonPlan, error) {
    plan, _ := uc.lessonPlanProvider.GetByID(planID)

    // Auth
    cs, _ := uc.courseSubjectProvider.GetByID(plan.CourseSubjectID)
    if cs.TeacherID != userID {
        return nil, errors.NewForbidden("only assigned teacher")
    }

    // Validar que tiene actividades
    if len(plan.Activities) == 0 {
        return nil, errors.NewValidation("plan must have activities before generating")
    }

    // Armar contexto
    ctx := uc.buildContext(plan, cs)

    // Llamar a Azure OpenAI
    proposal, err := uc.aiProvider.GenerateLessonProposal(ctx)
    if err != nil {
        return nil, errors.NewInternal("AI generation failed")
    }

    // Guardar
    plan.Proposal = &proposal
    uc.lessonPlanProvider.Update(plan)

    return uc.lessonPlanProvider.GetByID(planID)
}

func (uc *GenerateLessonProposalUsecase) buildContext(
    plan *entities.TeacherLessonPlan,
    cs *entities.CourseSubject,
) LessonPlanGenerationContext {
    // Get coordination document with sections
    doc, _ := uc.coordDocProvider.GetByID(plan.CoordinationDocumentID)

    // Get topics for this class
    classTopics := uc.getClassTopics(plan.CoordDocClassID)

    // Get shared class info
    isShared, sharedSubject, sharedTopics := uc.getSharedClassInfo(cs, plan.ClassNumber)

    // Build activity contexts
    activities := make([]ActivityContext, len(plan.Activities))
    for i, a := range plan.Activities {
        activities[i] = ActivityContext{
            Moment:      string(a.Moment),
            Name:        a.Activity.Name,
            Description: a.Activity.Description,
            Duration:    a.Activity.DurationMinutes,
            CustomNote:  stringOrEmpty(a.CustomNote),
        }
    }

    return LessonPlanGenerationContext{
        ClassNumber:            plan.ClassNumber,
        Title:                  plan.Title,
        Objective:              stringOrEmpty(plan.Objective),
        Topics:                 topicNames(classTopics),
        Activities:             activities,
        Source:                 buildSourceCtx(plan),
        MethodologicalStrategy: extractStrategy(doc),
        IsShared:               isShared,
        SharedWithSubject:      sharedSubject,
        SharedSubjectTopics:    sharedTopics,
    }
}
```

## Notas

- El usecase es idempotente: llamar varias veces sobreescribe la propuesta
- Si la generación falla, no se modifica el plan existente
- El contexto de clase compartida enriquece la propuesta para articulación interdisciplinaria
