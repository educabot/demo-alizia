# T-6.4.3: Tests

**Historia:** HU-6.4 — Contexto y enriquecimiento
**Tipo:** Tests
**Estado:** ⬜ Pendiente

---

## Descripcion

Tests unitarios para los context builders de coordinacion y planificacion. Verifican que el output contenga las secciones esperadas y que los campos vacios/nil se omitan correctamente. Se usan `strings.Contains` para verificar bloques clave (no matching exacto de strings, ya que los prompts pueden evolucionar).

## Implementacion

```go
// tests/ai/context_test.go
package ai_test

import (
    "strings"
    "testing"
    "time"

    "go-alizia/src/core/entities"
    "go-alizia/src/core/usecases/ai"
)

// --- Fixtures ---

func fullCoordDoc() *entities.CoordinationDocument {
    return &entities.CoordinationDocument{
        ID:        1,
        StartDate: time.Date(2026, 3, 1, 0, 0, 0, 0, time.UTC),
        EndDate:   time.Date(2026, 11, 30, 0, 0, 0, 0, time.UTC),
        Area:      &entities.Area{Name: "Ciencias Exactas"},
        Topics: []entities.Topic{
            {Name: "Pensamiento logico"},
            {Name: "Resolucion de problemas"},
        },
        Subjects: []entities.CoordDocSubject{
            {
                Subject:    entities.Subject{Name: "Matematicas"},
                ClassCount: 20,
                Topics: []entities.Topic{
                    {Name: "Pensamiento logico"},
                },
            },
            {
                Subject:    entities.Subject{Name: "Fisica"},
                ClassCount: 15,
                Topics: []entities.Topic{
                    {Name: "Resolucion de problemas"},
                },
            },
        },
        Sections: mustJSON(`{
            "methodological_strategy": {"value": "Aprendizaje basado en proyectos"},
            "problem_edge": {"value": "Integracion ciencias exactas"}
        }`),
    }
}

func coordDocWithoutSections() *entities.CoordinationDocument {
    doc := fullCoordDoc()
    doc.Sections = nil
    return doc
}

func sharedLessonPlanInput() ai.LessonPlanContextInput {
    return ai.LessonPlanContextInput{
        Plan: &entities.TeacherLessonPlan{
            Activities: []entities.LessonPlanActivity{
                {
                    Activity:    entities.Activity{Name: "Debate grupal", DurationMinutes: 15, Moment: "apertura"},
                    TeacherNote: "Usar ejemplos del barrio",
                },
                {
                    Activity: entities.Activity{Name: "Ejercicios practicos", DurationMinutes: 30, Moment: "desarrollo"},
                },
                {
                    Activity: entities.Activity{Name: "Puesta en comun", DurationMinutes: 10, Moment: "cierre"},
                },
            },
            SourceType:  "resource",
            SourceValue: "https://example.com/recurso.pdf",
        },
        CoordDocClass: &entities.CoordDocClass{
            ClassNumber: 3,
            Title:       "Introduccion a la mecanica",
            Objective:   "Comprender las leyes de Newton",
            Topics:      []entities.Topic{{Name: "Resolucion de problemas"}},
        },
        CoordDoc:     fullCoordDoc(),
        SharedWith:   &entities.Subject{Name: "Matematicas"},
        SharedTopics: []string{"Pensamiento logico", "Algebra basica"},
    }
}

func lessonPlanInputWithoutSource() ai.LessonPlanContextInput {
    input := sharedLessonPlanInput()
    input.Plan.SourceType = ""
    input.Plan.SourceValue = ""
    input.SharedWith = nil
    input.SharedTopics = nil
    return input
}

// --- Test cases ---

// 6.19: BuildCoordDocContext con documento completo → incluye area, topics, materias, secciones
func TestBuildCoordDocContext_FullDocument(t *testing.T) {
    result := ai.BuildCoordDocContext(fullCoordDoc())

    checks := []string{
        "Ciencias Exactas",
        "01/03/2026",
        "30/11/2026",
        "Pensamiento logico",
        "Resolucion de problemas",
        "Matematicas",
        "20 clases",
        "Fisica",
        "15 clases",
        "Aprendizaje basado en proyectos",
        "Integracion ciencias exactas",
    }
    for _, check := range checks {
        if !strings.Contains(result, check) {
            t.Errorf("expected output to contain %q, got:\n%s", check, result)
        }
    }
}

// 6.20: BuildCoordDocContext con secciones nil → no aparece bloque "Secciones"
func TestBuildCoordDocContext_NilSections(t *testing.T) {
    result := ai.BuildCoordDocContext(coordDocWithoutSections())

    if strings.Contains(result, "Secciones del documento") {
        t.Errorf("expected no 'Secciones del documento' block when sections is nil, got:\n%s", result)
    }
    // Pero si debe tener area, topics, materias
    if !strings.Contains(result, "Ciencias Exactas") {
        t.Errorf("expected area name in output, got:\n%s", result)
    }
}

// 6.21: BuildLessonPlanContext con clase compartida → incluye "Clase compartida" con materia y topics
func TestBuildLessonPlanContext_SharedClass(t *testing.T) {
    result := ai.BuildLessonPlanContext(sharedLessonPlanInput())

    checks := []string{
        "Clase compartida",
        "Matematicas",
        "Pensamiento logico",
        "Algebra basica",
        "Clase 3",
        "Introduccion a la mecanica",
    }
    for _, check := range checks {
        if !strings.Contains(result, check) {
            t.Errorf("expected output to contain %q, got:\n%s", check, result)
        }
    }
}

// 6.22: BuildLessonPlanContext sin fuente → no aparece bloque "Recurso" ni "Nota"
func TestBuildLessonPlanContext_NoSource(t *testing.T) {
    result := ai.BuildLessonPlanContext(lessonPlanInputWithoutSource())

    if strings.Contains(result, "Recurso anclado") {
        t.Errorf("expected no 'Recurso anclado' block, got:\n%s", result)
    }
    if strings.Contains(result, "Nota personalizada") {
        t.Errorf("expected no 'Nota personalizada' block, got:\n%s", result)
    }
    // Pero si debe tener la clase y actividades
    if !strings.Contains(result, "Clase 3") {
        t.Errorf("expected class info in output, got:\n%s", result)
    }
}

// 6.23: BuildLessonPlanContext actividades agrupadas correctamente por momento
func TestBuildLessonPlanContext_ActivitiesGroupedByMoment(t *testing.T) {
    result := ai.BuildLessonPlanContext(sharedLessonPlanInput())

    // Verificar que los 3 momentos aparecen
    moments := []string{"Apertura", "Desarrollo", "Cierre"}
    for _, m := range moments {
        if !strings.Contains(result, m) {
            t.Errorf("expected moment %q in output, got:\n%s", m, result)
        }
    }

    // Verificar orden: apertura antes que desarrollo, desarrollo antes que cierre
    aperturaIdx := strings.Index(result, "Apertura")
    desarrolloIdx := strings.Index(result, "Desarrollo")
    cierreIdx := strings.Index(result, "Cierre")

    if aperturaIdx >= desarrolloIdx {
        t.Errorf("expected Apertura before Desarrollo, got apertura=%d desarrollo=%d", aperturaIdx, desarrolloIdx)
    }
    if desarrolloIdx >= cierreIdx {
        t.Errorf("expected Desarrollo before Cierre, got desarrollo=%d cierre=%d", desarrolloIdx, cierreIdx)
    }

    // Verificar que la nota del docente aparece
    if !strings.Contains(result, "Usar ejemplos del barrio") {
        t.Errorf("expected teacher note in output, got:\n%s", result)
    }
}
```

## Notas

- Tests unitarios puros — sin dependencias externas
- Se usa `strings.Contains` en lugar de matching exacto, ya que el formato del contexto puede evolucionar
- Los fixtures (`fullCoordDoc`, `sharedLessonPlanInput`, etc.) representan escenarios realistas
- `mustJSON` es un helper que parsea JSON a `json.RawMessage` (panic si falla — solo para tests)
- El test 6.23 verifica el orden de los momentos comparando indices, no solo presencia
- Table-driven tests no se usan aqui porque cada caso verifica combinaciones diferentes de campos — los asserts son distintos por caso
