## Descripcion

Implementar `AzureOpenAIProvider` que satisface la interfaz `AIProvider` usando el SDK oficial de Azure para Go. Incluye retry con backoff exponencial, logging de tokens y el loop de function calling.

## Implementacion

// src/infra/ai/azure_openai.go
package ai

import (
    "context"
    "encoding/json"
    "fmt"
    "log/slog"
    "time"

    "github.com/Azure/azure-sdk-for-go/sdk/ai/azopenai"
    "github.com/Azure/azure-sdk-for-go/sdk/azcore/to"
    "go-alizia/src/core/providers"
)

const (
    maxToolIterations = 5
    defaultTimeout    = 30 * time.Second
    maxRetries        = 3
)

type AzureOpenAIProvider struct {
    client  *azopenai.Client
    model   string
    timeout time.Duration
}

// GenerateText envia system + user prompt y retorna texto libre.
func (p *AzureOpenAIProvider) GenerateText(ctx context.Context, systemPrompt, userPrompt string) (string, error) {
    ctx, cancel := context.WithTimeout(ctx, p.timeout)
    defer cancel()

    resp, err := p.completionWithRetry(ctx, []azopenai.ChatRequestMessageClassification{
        &azopenai.ChatRequestSystemMessage{Content: azopenai.NewChatRequestSystemMessageContent(systemPrompt)},
        &azopenai.ChatRequestUserMessage{Content: azopenai.NewChatRequestUserMessageContent(userPrompt)},
    }, nil, nil)
    if err != nil {
        return "", err
    }

    if len(resp.Choices) == 0 || resp.Choices[0].Message.Content == nil {
        return "", fmt.Errorf("ai: empty response from model")
    }

    return *resp.Choices[0].Message.Content, nil
}

// GenerateJSON usa response_format: json_schema para structured output.
func (p *AzureOpenAIProvider) GenerateJSON(ctx context.Context, systemPrompt, userPrompt string, schema interface{}) ([]byte, error) {
    ctx, cancel := context.WithTimeout(ctx, p.timeout)
    defer cancel()

    schemaBytes, err := json.Marshal(schema)
    if err != nil {
        return nil, fmt.Errorf("ai: invalid schema: %w", err)
    }

    responseFormat := &azopenai.ChatCompletionsJSONSchemaResponseFormat{
        JSONSchema: &azopenai.ChatCompletionsJSONSchema{
            Name:   to.Ptr("response"),
            Schema: schemaBytes,
            Strict: to.Ptr(true),
        },
    }

    resp, err := p.completionWithRetry(ctx, []azopenai.ChatRequestMessageClassification{
        &azopenai.ChatRequestSystemMessage{Content: azopenai.NewChatRequestSystemMessageContent(systemPrompt)},
        &azopenai.ChatRequestUserMessage{Content: azopenai.NewChatRequestUserMessageContent(userPrompt)},
    }, nil, responseFormat)
    if err != nil {
        return nil, err
    }

    if len(resp.Choices) == 0 || resp.Choices[0].Message.Content == nil {
        return nil, fmt.Errorf("ai: empty response from model")
    }

    raw := []byte(*resp.Choices[0].Message.Content)
    if !json.Valid(raw) {
        return nil, fmt.Errorf("ai: response is not valid JSON")
    }

    return raw, nil
}

// ChatWithTools maneja el loop completo de function calling.
func (p *AzureOpenAIProvider) ChatWithTools(ctx context.Context, req providers.ChatWithToolsRequest) (*providers.ChatWithToolsResponse, error) {
    ctx, cancel := context.WithTimeout(ctx, p.timeout)
    defer cancel()

    // 1. Convertir mensajes al formato Azure
    messages := convertMessages(req.Messages)

    // 2. Convertir tools al formato Azure
    tools := convertTools(req.Tools)

    var toolsUsed []string
    totalTokens := 0

    // 3. Loop de tool calls (max maxToolIterations)
    for i := 0; i < maxToolIterations; i++ {
        resp, err := p.completionWithRetry(ctx, messages, tools, nil)
        if err != nil {
            return nil, err
        }

        if resp.Usage != nil {
            totalTokens += int(*resp.Usage.TotalTokens)
        }

        choice := resp.Choices[0]

        // Si no hay tool calls, tenemos la respuesta final
        if len(choice.Message.ToolCalls) == 0 {
            content := ""
            if choice.Message.Content != nil {
                content = *choice.Message.Content
            }
            return &providers.ChatWithToolsResponse{
                Message:    content,
                ToolsUsed:  toolsUsed,
                TokensUsed: totalTokens,
            }, nil
        }

        // Agregar el mensaje del asistente con tool calls
        messages = append(messages, &azopenai.ChatRequestAssistantMessage{
            Content:   choice.Message.Content,
            ToolCalls: choice.Message.ToolCalls,
        })

        // Ejecutar cada tool call
        for _, tc := range choice.Message.ToolCalls {
            funcCall := tc.Function
            toolsUsed = append(toolsUsed, *funcCall.Name)

            result, err := req.ToolHandler(*funcCall.Name, *funcCall.Arguments)
            if err != nil {
                result = fmt.Sprintf("Error executing %s: %v", *funcCall.Name, err)
            }

            messages = append(messages, &azopenai.ChatRequestToolMessage{
                Content:    azopenai.NewChatRequestToolMessageContent(result),
                ToolCallID: tc.ID,
            })
        }
    }

    return nil, fmt.Errorf("ai: max tool iterations (%d) reached", maxToolIterations)
}

// completionWithRetry ejecuta la completion con retry y backoff exponencial para rate limits.
func (p *AzureOpenAIProvider) completionWithRetry(
    ctx context.Context,
    messages []azopenai.ChatRequestMessageClassification,
    tools []azopenai.ChatCompletionsToolDefinition,
    responseFormat azopenai.ChatCompletionsResponseFormatClassification,
) (azopenai.GetChatCompletionsResponse, error) {
    var lastErr error

    for attempt := 0; attempt < maxRetries; attempt++ {
        opts := azopenai.ChatCompletionsOptions{
            DeploymentName:      &p.model,
            Messages:            messages,
            MaxCompletionTokens: to.Ptr[int32](4096),
        }
        if len(tools) > 0 {
            opts.Tools = tools
        }
        if responseFormat != nil {
            opts.ResponseFormat = responseFormat
        }

        resp, err := p.client.GetChatCompletions(ctx, opts, nil)
        if err == nil {
            logTokenUsage(resp)
            return resp, nil
        }

        if isRateLimitError(err) {
            backoff := time.Duration(1<<attempt) * time.Second // 1s, 2s, 4s
            slog.Warn("ai: rate limited, retrying",
                "attempt", attempt+1,
                "backoff", backoff,
            )
            select {
            case <-time.After(backoff):
                lastErr = err
                continue
            case <-ctx.Done():
                return azopenai.GetChatCompletionsResponse{}, ctx.Err()
            }
        }

        return azopenai.GetChatCompletionsResponse{}, classifyError(err)
    }

    return azopenai.GetChatCompletionsResponse{}, fmt.Errorf("%w: %v", providers.ErrAIRateLimit, lastErr)
}

// logTokenUsage registra el uso de tokens para tracking de costos.
func logTokenUsage(resp azopenai.GetChatCompletionsResponse) {
    if resp.Usage == nil {
        return
    }
    slog.Info("ai: token usage",
        "prompt_tokens", *resp.Usage.PromptTokens,
        "completion_tokens", *resp.Usage.CompletionTokens,
        "total_tokens", *resp.Usage.TotalTokens,
    )
}

## Comportamiento

1. **`GenerateText`** — Arma system + user message, envia con retry, retorna el contenido de la primera choice
2. **`GenerateJSON`** — Igual pero con `response_format: json_schema` para garantizar structured output. Valida que la respuesta sea JSON valido
3. **`ChatWithTools`** — Loop:
   - Envia mensajes + tools al modelo
   - Si el modelo responde con `tool_calls` → ejecuta cada una via `ToolHandler`
   - Agrega resultados como mensajes `tool` y reenvia
   - Repite hasta respuesta sin tool calls o alcanzar `maxToolIterations`
4. **`completionWithRetry`** — Envuelve la llamada al SDK con retry exponencial (1s, 2s, 4s) solo para errores 429. Otros errores se retornan inmediatamente clasificados

## Notas

- Usar `max_completion_tokens` y NO `max_tokens` — requisito de `gpt-5-mini`
- NO setear `temperature` — usar el default del modelo
- El loop de tools tiene un cap de `maxToolIterations = 5` para prevenir loops infinitos
- `classifyError` mapea errores del SDK de Azure a los errores tipados definidos en `src/core/providers/errors.go`
- El backoff usa `select` con `ctx.Done()` para respetar cancelaciones del contexto
- `logTokenUsage` usa `slog` para que sea capturable por el sistema de logging de `team-ai-toolkit/applog`
