# T-3.2.4: Repository GORM

**Historia:** HU-3.2 — Áreas y materias
**Tipo:** Backend
**Estado:** ⬜ Pendiente

---

## Descripción

Implementar los repositories con GORM para áreas y materias. El listado de áreas debe hacer Preload de Subjects y Coordinators. La asignación de coordinador debe manejar el conflicto de unicidad retornando 409.

## Archivos

### `src/repositories/area.go`

```go
package repositories

import (
    "context"
    "errors"
    "src/core/entities"
    "src/core/providers"

    "github.com/jackc/pgx/v5/pgconn"
    "gorm.io/gorm"
)

type areaRepo struct {
    db *gorm.DB
}

func NewAreaRepository(db *gorm.DB) providers.AreaProvider {
    return &areaRepo{db: db}
}

func (r *areaRepo) Create(ctx context.Context, area *entities.Area) error {
    return r.db.WithContext(ctx).Create(area).Error
}

func (r *areaRepo) List(ctx context.Context, organizationID uint) ([]entities.Area, error) {
    var areas []entities.Area
    err := r.db.WithContext(ctx).
        Where("organization_id = ?", organizationID).
        Preload("Subjects").
        Preload("Coordinators").
        Find(&areas).Error
    return areas, err
}

func (r *areaRepo) Update(ctx context.Context, area *entities.Area) error {
    return r.db.WithContext(ctx).
        Model(area).
        Updates(map[string]interface{}{
            "name":        area.Name,
            "description": area.Description,
        }).Error
}

func (r *areaRepo) AssignCoordinator(ctx context.Context, areaID uint, userID uint) (*entities.AreaCoordinator, error) {
    ac := &entities.AreaCoordinator{
        AreaID: areaID,
        UserID: userID,
    }
    if err := r.db.WithContext(ctx).Create(ac).Error; err != nil {
        var pgErr *pgconn.PgError
        if errors.As(err, &pgErr) && pgErr.Code == "23505" {
            return nil, apperrors.ErrConflict // 409
        }
        return nil, err
    }
    return ac, nil
}
```

### `src/repositories/subject.go`

```go
package repositories

import (
    "context"
    "src/core/entities"
    "src/core/providers"

    "gorm.io/gorm"
)

type subjectRepo struct {
    db *gorm.DB
}

func NewSubjectRepository(db *gorm.DB) providers.SubjectProvider {
    return &subjectRepo{db: db}
}

func (r *subjectRepo) Create(ctx context.Context, subject *entities.Subject) error {
    return r.db.WithContext(ctx).Create(subject).Error
}

func (r *subjectRepo) List(ctx context.Context, organizationID uint) ([]entities.Subject, error) {
    var subjects []entities.Subject
    err := r.db.WithContext(ctx).
        Where("organization_id = ?", organizationID).
        Find(&subjects).Error
    return subjects, err
}
```

## Notas

- Siempre filtrar por `organization_id` en queries de listado para aislamiento multi-tenant
- Usar `Preload("Subjects")` y `Preload("Coordinators")` en el listado de áreas para cumplir los criterios de aceptación
- Detectar error de unique constraint (PostgreSQL code `23505`) y mapear a `apperrors.ErrConflict` para retornar 409 al cliente
- El package `repositories/` ya implica la implementación concreta — no hace falta sufijo `_postgres` en los archivos
- `Update` usa `Updates` con map para permitir setear `description` a null si se envía vacío
