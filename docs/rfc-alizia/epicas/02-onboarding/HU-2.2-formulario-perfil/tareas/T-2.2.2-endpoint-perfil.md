```go
func (r *userRepo) UpdateProfileData(ctx context.Context, userID uint, data map[string]interface{}) error {
    jsonData, err := json.Marshal(data)
    if err != nil {
        return err
    }
    return r.db.WithContext(ctx).
        Model(&entities.User{}).
        Where("id = ?", userID).
        Update("profile_data", gorm.Expr("?::jsonb", string(jsonData))).Error
}
```

## Usecase

### `src/core/usecases/onboarding/save_profile.go`

```go
package onboarding

import (
    "context"
    "fmt"
    "go-alizia/src/core/providers"
)

type SaveProfileUseCase struct {
    userProvider providers.UserProvider
    orgProvider  providers.OrganizationProvider
}

func NewSaveProfileUseCase(up providers.UserProvider, op providers.OrganizationProvider) *SaveProfileUseCase {
    return &SaveProfileUseCase{userProvider: up, orgProvider: op}
}

func (uc *SaveProfileUseCase) Execute(ctx context.Context, userID uint, orgID uint, data map[string]interface{}) error {
    // Obtener config de la org para validar campos requeridos
    org, err := uc.orgProvider.GetByID(ctx, orgID)
    if err != nil {
        return err
    }

    // Validar campos requeridos contra org config
    if err := validateProfileFields(org.Config, data); err != nil {
        return err
    }

    return uc.userProvider.UpdateProfileData(ctx, userID, data)
}
```

## Endpoints

### Actualizar handler

```go
// GET /users/me/profile
// Retorna el profile_data del usuario autenticado
func (h *UserHandler) GetProfile(ctx context.Context, r *web.Request) web.Response {
    user := r.AuthUser()
    return web.OK(user.ProfileData)
}

// PUT /users/me/profile
// Body: { "disciplines": [...], "years_of_experience": 12, ... }
// Valida contra la config de la org y guarda
func (h *UserHandler) SaveProfile(ctx context.Context, r *web.Request) web.Response {
    var data map[string]interface{}
    if err := r.Decode(&data); err != nil {
        return web.BadRequest("invalid body")
    }

    userID := r.AuthUserID()
    orgID := r.AuthOrgID()

    if err := h.saveProfileUC.Execute(ctx, userID, orgID, data); err != nil {
        return rest.HandleError(err)
    }
    return web.OK(data)
}
```

## Notas

- El profile data se reemplaza completo (PUT), no merge parcial — simplifica la lógica
- La validación de campos requeridos se hace en el usecase contra la config de la org
- El frontend puede leer el profile_data para pre-popular el formulario si el usuario vuelve
- El mismo endpoint sirve tanto para onboarding como para edición posterior en settings
