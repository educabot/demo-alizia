## Endpoint

### Actualizar `src/entrypoints/rest/admin/user_handler.go`

```go
// GET /users/me/onboarding-status
// Retorna { "completed": true/false, "completed_at": "2026-01-15T..." | null }
func (h *UserHandler) GetOnboardingStatus(ctx context.Context, r *web.Request) web.Response {
    user := r.AuthUser() // del JWT
    return web.OK(map[string]interface{}{
        "completed":    user.OnboardingCompletedAt != nil,
        "completed_at": user.OnboardingCompletedAt,
    })
}

// POST /users/me/onboarding/complete
// Idempotente: si ya completó, retorna 200 igual
func (h *UserHandler) CompleteOnboarding(ctx context.Context, r *web.Request) web.Response {
    userID := r.AuthUserID()
    if err := h.completeOnboardingUC.Execute(ctx, userID); err != nil {
        return rest.HandleError(err)
    }
    return web.OK(map[string]string{"status": "completed"})
}
```
